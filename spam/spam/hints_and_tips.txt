python3 -m flask shell
>>> from spam.models import Staff
>>> from spam.models import Location
>>> staff1 = Staff(name='ed', robot_id='fsdf')
>>> from spam.database import db_session
>>> db_session.add(staff1)
>>> Staff.query.all()
[]
>>> db_session.commit()
>>> Staff.query.all()
[<User 'ed'>]
>>> staff1.location
[]
>>> staff1.location = [Location(robot_location='robot', physical_location='physical')]
>>> staff1.location
[<Location 'physical'>]
>>> db_session.commit()
>>> db_session.query(Staff.id, Location.id).all()
[(1, 1)]
>>> db_session.query(Staff.id, Location.id, Location.staff_id).all()
[(1, 1, 1)]
>>> for u,a in db_session.query(Staff, Location).filter(Staff.id==Location.staff_id).all():
